program lacosDeRepeticao;

const numVezes = 10;
var entradaUsuario: string;
i: integer;

begin
    print { 'Este laço for irá executar ', numVezes, ' vezes' };

    for i := 5 to 10 do
        begin
            entradaUsuario := "aaaa";
            print { 'Número: ', i };
            i := i - 1;
        end;

    while i < 10 do
        begin
            read(entradaUsuario);
            print { 'Você digitou: ', entradaUsuario };
        end;
end.
