program lacosDeRepeticao;

const numVezes = 5;
var entradaUsuario: string;

begin
    print { 'Este laço for irá executar ', numVezes, ' vezes' };

    for i := 1 to 5+3 do
        begin
            print { 'Número: ', i };
            i := i + 1;
        end;

    while i = 5 do
        begin
            read(entradaUsuario);
            print { 'Você digitou: ', entradaUsuario };
        end;
end.
